//Let v[N] be a sorted array of N distinct elements and let k be a key.

#include <stdio.h>

int BinarySearch(int v[],int N ,int k){
    int found,l=0,r=N-1,m;
    while(l<=r && found ==0){
        m = (l+r)/2;
        if(v[m] == k)
            found = 1;
        else if (v[m]<k)
            l = m+1;
        else
            r = m-1;
    }
    if(found == 0)
        return -1;
    else
        return m;
}

int main(){
    int v[4] = {6,8,0,9},N=sizeof(v)/sizeof(v[0]),k=9;
    int index = BinarySearch(v,N,k);
    printf("%d",index);
}
