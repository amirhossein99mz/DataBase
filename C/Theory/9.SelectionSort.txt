void SelectionSort(int A[],int N){
    int i,j,l=0,r=N-1,min,temp;
    for(i=l;i<r;i++){
        min =i;
        for(j=i+1;j<=r;j++){
            if(A[j]<A[min])
                min = j;
        }
        if (min!=i){
            temp = A[i];
            A[i] = A[min];
            A[min] = temp;
        }
     }
     return;
}
