void merge(int A[],int B[],int l, int q,int r){
int i,j,k;
i =l;
j =q+1;
for(k=l;k<=r;k++){
    if(i>q)
        B[k] = A[j++];
    else if(j>r)
        B[k] = A[i++];
    else if (A[i]<A[j]) || (A[i] == A[j])
        B[k] = A[i++];
    else
        B[k] = A[i++];
for (l=l;k<=r;k++)
    A[k] = B[k];
return;
}
